这一题没有一遍过说明了一个缺陷：在得到正确的转移方程之后，对于初始化不慎重

一开始的错误代码：
```C++
class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n + 1, 1e5);
        dp[1] = 1;
        for (int num = 2; num <= n; ++num)
        {
            for (int i = 1; i * i <= num; ++i) dp[num] = min(dp[num], dp[num - i * i] + 1);
        }
        return dp[n];
    }
};
```

正确代码：
```C++
class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n + 1, 1e5);
        dp[0] = 0;
        for (int num = 1; num <= n; ++num)
        {
            for (int i = 1; i * i <= num; ++i) dp[num] = min(dp[num], dp[num - i * i] + 1);
        }
        return dp[n];
    }
};
```
